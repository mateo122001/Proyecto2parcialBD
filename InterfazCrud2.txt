package sql;
import java.sql.Connection;
import java.sql.DriverManager;
import javax.swing.JOptionPane;

public class conexionsql {
    Connection conn =null;
    String url="jdbc:postgresql://localhost:5432/aseguradoraP";
    String usuario="postgres";
    String clave="321108sm";
    public Connection conectar(){
        try{
            Class.forName("org.postgresql.Driver");  // esto es para que los drivers funcionen 
            conn=DriverManager.getConnection(url,usuario,clave);
        }catch(Exception e){
            JOptionPane.showMessageDialog(null, "Error al conectar"+e,"Error",JOptionPane.ERROR_MESSAGE);
        }
        return conn;
    
}
}

package sql;

import getset.variables;

public class crudsql extends conexionsql {

    java.sql.Statement st;
    ResultSet rs;
    variables var=new variables();
    public void insertar(String ida, String nombre, String apellido, String cedula,
    String Fnacimiento, String direccion, String telefono, String contratacion) {
        try {
            Connection conexion = conectar();
            st = conexion.createStatement(); // esta linea nos permite ejecutar sentencias sql
            String sql = "insert into Empleado(Id_Aseguradora,Nombre_Empleado,Apellido_Empleado,"
            + "Cedula_Empleado,FechaNacimiento_Empleado,Direccion_Empleado,Telefono_Empleado,FechaInicioTrabajo_Empleado) "
            + "values('" + ida + "','" + nombre + "','" + apellido + "','" + cedula + "','" + Fnacimiento + "','" + direccion + "',"
            + "'" + telefono + "','" + contratacion + "');";
            st.execute(sql);
            st.close(); 
            conexion.close();
            JOptionPane.showMessageDialog(null, "Ingreso exitoso", "Message", JOptionPane.INFORMATION_MESSAGE);
        } catch (Exception e) {
            JOptionPane.showMessageDialog(null, "El registro no se guardó" + e, "Message", JOptionPane.ERROR_MESSAGE);
        }
        
    }public void mostrar(String idempleado){
        try{
            Connection conexion= conectar();
               st = conexion.createStatement();
               String sql="select * from Empleado where Id_Empleado='"+idempleado+"'";
               rs=st.executeQuery(sql);
               if(rs.next()){
                   
                   var.setIdempleado(rs.getString("Id_Empleado"));
                   var.setIdaseguradora(rs.getString("Id_Aseguradora"));
                   var.setNombre(rs.getString("Nombre_Empleado"));
                   var.setApellido(rs.getString("Apellido_Empleado"));
                   var.setCedula(rs.getString("Cedula_Empleado"));
                   var.setFnacimiento(rs.getString("FechaNacimiento_Empleado"));
                   var.setDireccion(rs.getString("Direccion_Empleado"));
                   var.setContratacion(rs.getString("fechaInicioTrabajo_Empleado"));
                   var.setTelefono(rs.getString("Telefono_Empleado"));
                   
               }else {
                    var.setIdempleado("");
                   var.setIdaseguradora("");
                   var.setNombre("");
                   var.setApellido("");
                   var.setCedula("");
                   var.setFnacimiento("");
                   var.setDireccion("");
                   var.setContratacion("");
                   var.setTelefono("");
                   JOptionPane.showMessageDialog(null,"No se encontró registro","Sin registro",JOptionPane.INFORMATION_MESSAGE);
               }
               st.close();
               conexion.close();
        }catch(Exception e){
         JOptionPane.showMessageDialog(null,"Error en el sistema de busqueda","Error busqueda",JOptionPane.ERROR_MESSAGE);
        }
    }
}
